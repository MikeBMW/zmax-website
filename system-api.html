引擎切换API（Sys-1统一框架）


model.set_engine('act')      # 本地 ACT 52M · 1.2ms
model.set_engine('vtla')     # 远程 VTLA 450M · 280ms (4090:50052)
model.set_engine('groot')    # 远程 GR00T 2B · ~500ms (4090:50052)
model.set_engine('smolvla')  # 本地 smolvla 450M · 215ms (Sys-11)
model.set_engine('lew')      # 本地 smolvla_lew 628M · 186ms (Sys-12)
# 远程不可达 → 自动回退ACT
